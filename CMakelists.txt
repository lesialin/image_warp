cmake_minimum_required(VERSION 3.10)
project( image_warp )
if(MSVC)
    add_definitions(-D_CRT_SECURE_NO_WARNINGS)
endif()

find_package( OpenCV REQUIRED )
include_directories( ${OpenCV_INCLUDE_DIRS} )
include_directories(${PROJECT_SOURCE_DIR})
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY $<1:${CMAKE_SOURCE_DIR}/prebuilt>)
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY  $<1:${CMAKE_SOURCE_DIR}/prebuilt>)
add_executable( image_warp  image_warping.cpp main.cpp )
target_link_libraries( image_warp ${OpenCV_LIBS} )
